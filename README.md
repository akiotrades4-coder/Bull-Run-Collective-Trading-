function connectWS(){
  try{
    ws=new WebSocket(API.replace('http','ws'));
    ws.onopen=()=>addLog(ts(),'info','Connected to server at '+API);
    ws.onmessage=e=>{try{handleEv(JSON.parse(e.data))}catch(_){}};
    ws.onclose=()=>{addLog(ts(),'warn','Server disconnected — retrying...');setTimeout(connectWS,3000)};
    ws.onerror=()=>addLog(ts(),'warn','Cannot reach '+API+' — check npm start is running');
  }catch(e){addLog(ts(),'warn','WS error: '+e.message)}
}
function handleEv(d){
  if(d.event==='connected'){(d.sessions||[]).forEach(b=>setBroker(b,true))}
  else if(d.event==='auto_trade'){positions.push({...d.position,fromSrv:true});addLog(ts(),'auto','[AUTO] '+d.position.side+' '+d.position.qty+'x '+d.position.symbol+' via '+d.position.broker);updatePosUI();updateMetrics()}
  else if(d.event==='position_closed'){positions=positions.filter(p=>p.id!==d.position.id);realPnl+=d.pnl;if(d.pnl>0)wins++;else losses++;addLog(ts(),'auto','[EXIT] '+d.position.symbol+' → '+fmtP(d.pnl));updatePosUI();updateMetrics()}
  else if(d.event==='engine_log'){addEngLog(d.ts,d.type,d.msg)}
  else if(d.event==='engine_stopped'){setAutoUI(false)}
}

async function connect(broker){
  const btn=document.getElementById('btn-'+broker);btn.textContent='Connecting...';btn.disabled=true;
  showMsg(broker,'Authenticating...','sm-loading');
  let creds={};
  if(broker==='tradovate')creds={username:document.getElementById('tv-user').value,password:document.getElementById('tv-pass').value,apiKey:document.getElementById('tv-key').value,env:document.getElementById('tv-env').value};
  else if(broker==='ninjatrader')creds={apiKey:document.getElementById('nt-key').value,accountId:document.getElementById('nt-acct').value,env:document.getElementById('nt-env').value};
  else if(broker==='projectx')creds={username:document.getElementById('px-user').value,apiKey:document.getElementById('px-key').value};
  else if(broker==='webull')creds={appId:document.getElementById('wb-id').value,appSecret:document.getElementById('wb-secret').value};
  else if(broker==='coinbase')creds={keyName:document.getElementById('cb-keyname').value,privateKey:document.getElementById('cb-privkey').value,permission:'trade'};
  try{
    const res=await apiPost('/api/auth/'+broker,creds);
    if(res.ok){setBroker(broker,true);showMsg(broker,'Connected ✓ — '+(res.account?.account||'OK'),'sm-ok');addLog(ts(),'info',broker+' authenticated ✓')}
    else{showMsg(broker,'Error: '+res.error,'sm-err');btn.textContent='Connect';btn.disabled=false}
  }catch(e){showMsg(broker,'Cannot reach server — is npm start running?','sm-err');btn.textContent='Connect';btn.disabled=false}
}
function setBroker(b,on){
  connected[b]=on;
  const tag=document.getElementById('bt-'+b);const badge=document.getElementById('badge-'+b);const btn=document.getElementById('btn-'+b);
  if(tag)tag.classList.toggle('on',on);
  if(badge){badge.textContent=on?'Connected':'Disconnected';badge.className='cbadge '+(on?'cbon':'cboff')}
  if(btn){btn.textContent=on?'Disconnect':'Connect';btn.disabled=false;if(on){btn.onclick=()=>discBroker(b);btn.classList.add('disc')}else{btn.onclick=()=>connect(b);btn.classList.remove('disc')}}
}

async function placeOrder(side){
  const sym=document.getElementById('oi').value,broker=document.getElementById('obroker').value;
  const qty=parseInt(document.getElementById('oqty').value)||1,type=document.getElementById('otype').value;
  const sl=parseInt(document.getElementById('osl').value)||8,tp=parseInt(document.getElementById('otp').value)||16;
  const price=INSTS[sym]?.price||21000,tick=0.25;
  const msgEl=document.getElementById('order-msg');msgEl.textContent='Sending...';msgEl.className='smsg sm-loading';msgEl.style.display='block';
  if(side==='FLAT'){try{const r=await apiPost('/api/flatten',{broker,symbol:sym});msgEl.textContent=r.ok?'Flattened':'Error: '+r.error;msgEl.className='smsg '+(r.ok?'sm-ok':'sm-err')}catch(e){msgEl.textContent='Server unreachable';msgEl.className='smsg sm-err'}return}
  const slP=side==='BUY'?price-sl*tick:price+sl*tick,tpP=side==='BUY'?price+tp*tick:price-tp*tick;
  try{
    const r=await apiPost('/api/order',{broker,symbol:sym,side,qty,type,stopLoss:slP,takeProfit:tpP});
    if(r.ok){msgEl.textContent=side+' placed — '+r.orderId;msgEl.className='smsg sm-ok';addLog(ts(),side.toLowerCase(),side+' '+qty+'x '+sym+' @ '+fmt(price)+' via '+broker);positions.push({symbol:sym,side,qty,entry:price,broker,id:r.orderId,fromSrv:true});updatePosUI();updateMetrics()}
    else{msgEl.textContent='Error: '+r.error;msgEl.className='smsg sm-err'}
  }catch(e){msgEl.textContent='Server unreachable';msgEl.className='smsg sm-err'}
}

function updateMetrics(){
  let open=0;positions.forEach(p=>{open+=(INSTS[p.symbol]?.price-(p.entry||0))*(p.side==='BUY'?1:-1)*(p.qty||1)*(INSTS[p.symbol]?.pv||1)});
  const net=realPnl+open,total=wins+losses;
  const set=(id,v,c)=>{const el=document.getElementById(id);if(el){el.textContent=v;el.className='mv '+c}};
  set('m-net',fmtP(net),pCls(net));set('m-real',fmtP(realPnl),pCls(realPnl));set('m-open',fmtP(open),pCls(open));set('m-wr',total?Math.round(wins/total*100)+'%':'—',total?pCls(wins/total-.5):'neu');
  const dll=parseInt(document.getElementById('rs-dll')?.value)||500,mc=parseInt(document.getElementById('rs-mc')?.value)||5;
  const loss=Math.max(0,-realPnl),contracts=positions.reduce((s,p)=>s+(p.qty||1),0);
  const lr=Math.min(loss/dll,1),cr=Math.min(contracts/mc,1);
  const rh='<div class="rrow"><div class="rlbl"><span>Daily loss</span><span>$'+loss.toFixed(0)+' / $'+dll+'</span></div><div class="rtrack"><div class="rfill" style="width:'+Math.round(lr*100)+'%;background:'+(lr>.7?'var(--red)':'var(--green)')+'"></div></div></div><div class="rrow"><div class="rlbl"><span>Contracts</span><span>'+contracts+' / '+mc+'</span></div><div class="rtrack"><div class="rfill" style="width:'+Math.round(cr*100)+'%;background:var(--blue)"></div></div></div>';
  ['risk-bars','rc-risk'].forEach(id=>{const el=document.getElementById(id);if(el)el.innerHTML=rh});
  const rd=document.getElementById('r-dll');if(rd){rd.textContent='$'+(dll-loss).toFixed(0);rd.className='mv '+(lr>.7?'neg':'pos')}
  const rcon=document.getElementById('r-con');if(rcon)rcon.textContent=contracts+'/'+mc;
  const rdd=document.getElementById('r-dd');if(rdd)rdd.textContent=(lr*100).toFixed(1)+'%';
}
